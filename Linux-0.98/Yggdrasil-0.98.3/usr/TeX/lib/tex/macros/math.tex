%------------------------------------------------------
% Make active characters from 128 (0x80) to 255 (0xff)
%------------------------------------------------------
\newcount\asciiChar
\asciiChar=128
\loop
  \catcode\asciiChar=\active
  \advance\asciiChar by1
  \ifnum\asciiChar<256
\repeat
